179	const shardCache = new Map<number, Promise<DetailRecord[]>>();
180	
181	export function fetchShard(manifest: Manifest, shard: number): Promise<DetailRecord[]> {
182	  let p = shardCache.get(shard);
183	  if (!p) {
184	    const name = `details/shard-${String(shard).padStart(3, "0")}.json`;
185	    // Cache-bust like the other artifacts: detail shards are rebuilt in lockstep with
186	    // nodes-text.json, so its content hash is a correct version for every shard. Without
187	    // this, a rebuilt dataset serves stale shards -> a node shows another paper's details.
188	    const h = manifest.hashes?.["nodes-text.json"];
189	    const q = h ? `?v=${h}` : "";
190	    p = fetch(`${BASE}/${name}${q}`).then((r) => {
191	      if (!r.ok) throw new Error(`shard ${shard}: HTTP ${r.status}`);
192	      return r.json();
193	    });
194	    p.catch(() => shardCache.delete(shard));
195	    shardCache.set(shard, p);
196	  }
197	  return p;
198	}
